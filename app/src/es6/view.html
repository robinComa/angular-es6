Inheritance

class Animal {
    constructor(name) {
        this.name = name;
    }
    toString() {
        return `{Animal : ${this.name}}`;
    }
}
class Dog extends Animal {
    constructor(name) {
        super(name);
    }
    toString() {
        return `{Dog : ${this.name}}`;
    }
}
var animal = new Animal('An animal')
var dog = new Dog('Bill');

console.log(`animal : ${animal}`);
console.log(`dog : ${dog}`);
